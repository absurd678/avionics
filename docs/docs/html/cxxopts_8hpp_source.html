669 inline OptionNames split_option_names(const std::string &text)
670 {
671  OptionNames split_names;
672 
673  std::string::size_type token_start_pos = 0;
674  auto length = text.length();
675 
676  if (length == 0)
677  {
678  throw_or_mimic<exceptions::invalid_option_format>(text);
679  }
680 
681  while (token_start_pos < length) {
682  const auto &npos = std::string::npos;
683  auto next_non_space_pos = text.find_first_not_of(' ', token_start_pos);
684  if (next_non_space_pos == npos) {
685  throw_or_mimic<exceptions::invalid_option_format>(text);
686  }
687  token_start_pos = next_non_space_pos;
688  auto next_delimiter_pos = text.find(',', token_start_pos);
689  if (next_delimiter_pos == token_start_pos) {
690  throw_or_mimic<exceptions::invalid_option_format>(text);
691  }
692  if (next_delimiter_pos == npos) {
693  next_delimiter_pos = length;
694  }
695  auto token_length = next_delimiter_pos - token_start_pos;
696  // validate the token itself matches the regex /([:alnum:][-_[:alnum:]]*/
697  {
698  const char* option_name_valid_chars =
699  "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
700  "abcdefghijklmnopqrstuvwxyz"
701  "0123456789"
702  "_-.?";
703 
704  if (!std::isalnum(text[token_start_pos], std::locale::classic()) ||
705  text.find_first_not_of(option_name_valid_chars, token_start_pos) < next_delimiter_pos) {
706  throw_or_mimic<exceptions::invalid_option_format>(text);
707  }
708  }
709  split_names.emplace_back(text.substr(token_start_pos, token_length));
710  token_start_pos = next_delimiter_pos + 1;
711  }
712  return split_names;
713 }

743  }

941 template <typename T>
942 void
943 integer_parser(const std::string& text, T& value)
944 {
945  parser_tool::IntegerDesc int_desc = parser_tool::SplitInteger(text);
946 
947  using US = typename std::make_unsigned<T>::type;
948  constexpr bool is_signed = std::numeric_limits<T>::is_signed;
949 
950  const bool negative = int_desc.negative.length() > 0;
951  const uint8_t base = int_desc.base.length() > 0 ? 16 : 10;
952  const std::string & value_match = int_desc.value;
953 
954  US result = 0;
955 
956  for (char ch : value_match)
957  {
958  US digit = 0;
959 
960  if (ch >= '0' && ch <= '9')
961  {
962  digit = static_cast<US>(ch - '0');
963  }
964  else if (base == 16 && ch >= 'a' && ch <= 'f')
965  {
966  digit = static_cast<US>(ch - 'a' + 10);
967  }
968  else if (base == 16 && ch >= 'A' && ch <= 'F')
969  {
970  digit = static_cast<US>(ch - 'A' + 10);
971  }
972  else
973  {
974  throw_or_mimic<exceptions::incorrect_argument_type>(text);
975  }
976 
977  US limit = 0;
978  if (negative)
979  {
980  limit = static_cast<US>(std::abs(static_cast<intmax_t>((std::numeric_limits<T>::min)())));
981  }
982  else
983  {
984  limit = (std::numeric_limits<T>::max)();
985  }
986 
987  if (base != 0 && result > limit / base)
988  {
989  throw_or_mimic<exceptions::incorrect_argument_type>(text);
990  }
991  if (result * base > limit - digit)
992  {
993  throw_or_mimic<exceptions::incorrect_argument_type>(text);
994  }
995 
996  result = static_cast<US>(result * base + digit);
997  }
998 
999  detail::check_signed_range<T>(negative, result, text);
1000 
1001  if (negative)
1002  {
1003  checked_negate<T>(value, result, text, std::integral_constant<bool, is_signed>());
1004  }
1005  else
1006  {
1007  value = static_cast<T>(result);
1008  }
1009 }
